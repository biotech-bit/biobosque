// --- CONFIGURAÇÃO INICIAL E BANCO --- var map = L.map('mapa').setView([-4.2238, -55.9877], 18); L.tileLayer('http://{s}.google.com/vt/lyrs=s&x={x}&y={y}&z={z}', { maxZoom: 20, subdomains:['mt0','mt1','mt2','mt3'] }).addTo(map); var db = JSON.parse(localStorage.getItem('bio_vTCC_PRO')) || []; var lat, lng; // Captura clique no mapa map.on('click', e => { lat = e.latlng.lat; lng = e.latlng.lng; alert("📍 Localização capturada!"); }); // Função para salvar com segurança function salvar() { const nome = document.getElementById('p-nome').value; if(!nome || !lat) { alert("⚠️ ATENÇÃO: Marque a localização no mapa e digite o nome!"); return; } const novaArvore = { id: Date.now(), nome: nome, esp: document.getElementById('p-esp').value, fam: document.getElementById('p-fam').value, ord: document.getElementById('p-ord').value, cla: document.getElementById('p-cla').value, obs: document.getElementById('p-obs').value, lat: lat, lng: lng }; db.push(novaArvore); localStorage.setItem('bio_vTCC_PRO', JSON.stringify(db)); alert("✅ Árvore catalogada com sucesso!"); // Limpa campos e recarrega lista document.querySelectorAll('input, textarea').forEach(i => i.value = ""); render(); } // Renderiza os pontos no mapa e a lista function render() { map.eachLayer(l => { if(l instanceof L.Marker) map.removeLayer(l); }); const list = document.getElementById('lista-plantas'); list.innerHTML = ""; db.forEach(item => { L.marker([item.lat, item.lng]).addTo(map).bindPopup(item.nome); list.innerHTML += `
${item.nome}
${item.esp}
`; }); } // Inicializa window.onload = render;
